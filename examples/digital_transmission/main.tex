%------------ Information Theory ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.45\textwidth}
        $m_i$ only on leaves $\Rightarrow$ prefix code (instantaneous and uniquely decodable)\\
        Not prefix code $\Rightarrow$ not inst.\\
        $m_i + m_j = m_k \Rightarrow$ not uniq. dec.\\
        Natural binary ($\# \mathcal{A} = M$): $\lceil \log_2(M) \rceil$ [bits/symbol]\\
        $r_b = r_s \lceil \log_2(M) \rceil \qquad \mathrm{Redundancy} = D - H(\mathcal{A}) \qquad D = \log_2(M)$\\
        $I(m_i) = \log_2(1/p_i)$ [bits]\\
        $H(\mathcal{A}) = \sum_{i = 1}^M p_iI(m_i)$ [bits/symbol]\\
        Info bit rate: $r_s H(\mathcal{A}) \leq r_b$ [bits/s]\\
        Average word length: $\overline{L} = \sum_{i}^M p_i \ell_i$ [bits/symbol]\\
        Efficiency: $\eta = \frac{H(\mathcal{A})}{\overline{L}}$\\
        $T_\mathrm{min} = \frac{\#\mathrm{src}_\mathrm{symb} \times L_\mathrm{min}}{r_b} \qquad T_\mathrm{max} = \frac{\#\mathrm{src}_\mathrm{symb} \times L_\mathrm{max}}{r_b} \qquad T_\mathrm{av} = \frac{\#\mathrm{src}_\mathrm{symb} \times \overline{L}}{r_b}$\\
        $P(e) = \sum_{i}P(x_i) (1 - p_{ii})$\\
        $H(Y|X) = \sum_{i = 1}^M\sum_{j=1}^N P(x_i, y_j) \log_2\left(\frac{1}{P(y_j | x_i)}\right) = H(X, Y) - H(X)$\\
        Mutual info: $I(X, Y) = H(X) - H(X|Y) = H(Y) - H(Y|X) = \sum_{i = 1}^M\sum_{j=1}^N P(x_i, y_j)\log_2\left[\frac{P(x_i, y_j)}{P(x_i) P(y_j)}\right]$\\
        BSC: $H(p) = p\log_2\left(\frac{1}{p}\right) + (1-p)\log_2\left(\frac{1}{1-p}\right)$\\
        BSC: $C = 1 - H(p)$, BEC: $C = 1 - \varepsilon$, Generalized: $C = \mathrm{max} \ I(X, Y)$ [bit/transmission]\\
        Channel cap.: $r_b < C = B \log_2\left(1 + \frac{P}{N_0 B}\right) = B \log_2\left(1 + \mathrm{SNR}\right) \Rightarrow B_\mathrm{min} = \frac{r_b}{\log_2(1 + \mathrm{SNR})}$
    \end{minipage}
};
\node[fancytitle, right=10pt] at (box.north west) {\fontfamily{cmss}\selectfont 1. Information Theory};
\end{tikzpicture}

%------------ Bandpass Communication ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.8\textwidth}
        \textbf{Bandpass Process:} $s(t) = \mathrm{Re}\left\{z(t)e^{j2 \pi f_c t}\right\} \qquad z(t) = I(t) + jQ(t)$\\
        $s(t) = A(t)\cos\left[2\pi f_c t + \phi(t)\right] = \underbrace{A(t)\cos(\phi)}_{I(t)}\cos(2\pi f_c t) - \underbrace{A(t)\sin(\phi)}_{Q(t)}\sin(2\pi f_c t)$\\
        $A(t) = \sqrt{I^2(t) + Q^2(t)} \qquad \phi(t) = \mathrm{arctan}_2(Q(t), I(t))$\\
        $G_s(f) = \frac{1}{4}\left[G_z(-f-f_c) + G_z(f-f_c)\right]$\\
        \noindent\makebox[\linewidth]{\rule{\textwidth}{0.8pt}}
        $M-$\textbf{PAM:} $G_s(f) = \frac{d^2(M^2-1)}{12T_s} \left[\left|G(f + f_c)\right|^2 + \left|G(f - f_c)\right|^2\right] \qquad \overline{P}_\mathrm{PAM} = \frac{d^2(M^2 - 1)}{6T_s}E_g \qquad P_b = \frac{2(M-1)}{M\log_2(M)}Q\left(\sqrt{\frac{6\overline{E}_b\log_2(M)}{(M^2-1)N_0}}\right)$\\

        $M-$\textbf{PSK:} $G_s(f) = \frac{A^2}{4T_s} \left[\left|G(f + f_c)\right|^2 + \left|G(f - f_c)\right|^2\right] \qquad \overline{P}_\mathrm{PSK} = \frac{A^2}{2T_s}E_g \qquad P_b = \frac{2}{\log_2(M)}Q\left[\sqrt{\frac{2E_b\log_2(M)}{N_0}}\sin\left(\frac{\pi}{M}\right)\right]$\\

        $M-$\textbf{QAM:} $G_s(f) = \frac{d^2(M - 1)}{6T_s} \left[\left|G(f + f_c)\right|^2 + \left|G(f - f_c)\right|^2\right] \qquad \overline{P}_\mathrm{QAM} = \frac{d^2(M - 1)}{3T_s}E_g \qquad P_b = \frac{4}{\log_2(M)}\left(1 - \frac{1}{\sqrt{M}}\right)Q\left(\sqrt{\frac{3\overline{E}_b\log_2(M)}{(M-1)N_0}}\right)$\\

        $M$\textbf{SK:} $G_z(f) = \frac{16A^2T_b}{\pi^2} \left[\frac{\cos(2\pi f T_b)}{1 - 16f^2T_b^2}\right]^2 \qquad P_b = \frac{M}{2}Q\left(\sqrt{\frac{E_b}{N_0} \log_2(M)}\right)$ \big| $\underbrace{\textbf{Orthogonal mod.:}}_{\textbf{coherent FSK, ASK}} \ P_b = (M-1)Q\left(\sqrt{\frac{E_b\log_2(M)}{N_0}}\right) \approx Q\left(\sqrt{\frac{E_b}{N_0}}\right)$\\

        \vspace{-0.4cm}
        
        Union Bound: $P_s \leq \frac{1}{M}\sum_{k=0}^{M-1}\sum_{i = 0, \ i \neq k}^{M-1} Q\left(\frac{d_{ik}}{\sqrt{2N_0}}\right)$\\
        $M-\mathrm{PSK}$: $s_k = (s_{k_0}, s_{k_1}) = \left(\sqrt{E_s}\cos\left(\frac{2\pi k}{M}\right), \sqrt{E_s}\sin\left(\frac{2\pi k}{M}\right)\right)$ \big| $\textbf{non-coherent FSK, ASK:} \ P_b = \frac{1}{2}\exp\left(-\frac{E_b}{2N_0}\right)$\\
        Orthogonal Modulation: $P_b = Q\left(\sqrt{\frac{E_b}{N_0}}\right) \qquad \mathbb{E}\left\{w(t)w(\lambda)\right\} = \frac{N_0}{2}\delta(t-\lambda)$\\
        BPSK with phase jitter: $P_b = Q\left(\sqrt{\frac{2E_b}{N_0}}\cos \theta\right)$
    \end{minipage}
};

%------------ Channel Coding ---------------
\begin{tikzpicture}
\node [mybox] (box){%
    \begin{minipage}{0.45\textwidth}
        Codeword ($c$), Message word ($m$), Generator matrix ($G$), Detected words ($r$), Parity bit matrix ($P$), Parity-check matrix ($H$), Syndrome ($S$), Error pattern ($e$), Corrected codeword ($\hat{c}$)\\
        Minimum Hamming distance between 2 codewords $d_\mathrm{min}$\\
        $t$ error detections $\Leftrightarrow d_\mathrm{min} \geq t + 1$; $t$ error corrections $\Leftrightarrow d_\mathrm{min} \geq 2t + 1$\\
        $(n, k) = (\#\mathrm{bits \ codeword}, \ \#\mathrm{bits \ message word})$\\
        $c = m G: (k \times n) = (k \times k) \times (k \times n)$\\
        $G = [P | I_k] : (k \times n) = (k \times (n - k)) \times (k \times k)$\\
        $H = [I_{(n-k)} | P^T] : ((n - k) \times n) = ((n - k) \times (n - k)) \times ((n - k) \times k)$\\
        $S = eH^T$, $e$ all error patterns\\
        $S = rH^T \rightarrow \hat{e} \rightarrow \hat{c} = r + \hat{e} \rightarrow \hat{m}$\\
        Uncoded sys.: $P_b = Q\left(\sqrt{\frac{2E_b}{N_0}}\right)$\\
        Coded sys at detector output: $\Tilde{P}_b = Q\left(\sqrt{\frac{2\Tilde{E}_b}{N_0}}\right)$, $\Tilde{E}_b = \frac{k}{n}E_b$\\
        Coded sys. at decoder output (word): $\Tilde{P}_E \leq 1 - \sum_{i = 0}^t {n \choose i} \Tilde{P}_b^i\left(1 - \Tilde{P}_b\right)^{n-i}$\\
        Coded sys. at decoder output (bit): $P_{BD} \approx \frac{2t+1}{n} \Tilde{P}_E$\\
        Convolutional coding: $(n, k, L)$, $n$ output, $k$ input, $L$ registers\\
        \# states $= 2^{k(L-1)}$\\
        N/Y conv. cod. -- Hard (Hamming distance): $G_a = 10\log\left(\frac{k}{n}D_f / 2\right)$; Soft (Euclidean distance): $G_a = 10\log\left(\frac{k}{n}D_f\right)$
    \end{minipage}
